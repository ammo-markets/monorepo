**mint.ts -- handleMintFinalized:**

After the existing `updateMany` that sets `status: "COMPLETED"` and `tokenAmount`, also set `usdcAmount` if it was somehow missing. More importantly, add **self-healing**: if `updateMany` returns `count === 0` (no matching pending order), create the order from the finalization event directly.

```typescript
export async function handleMintFinalized(
  tx: PrismaTx,
  args: MintFinalizedArgs,
  meta: EventMeta,
): Promise<void> {
  const caliber = addressToCaliber(meta.address);
  const prismaCaliber = CALIBER_TO_PRISMA[caliber] as PrismaCaliber;

  const { count } = await tx.order.updateMany({
    where: {
      onChainOrderId: args.orderId.toString(),
      caliber: prismaCaliber,
      type: "MINT",
      status: "PENDING",
    },
    data: {
      status: "COMPLETED",
      tokenAmount: args.tokenAmount.toString(),
    },
  });

  // Self-healing: if no pending order exists, create from finalization event
  if (count === 0) {
    const userAddress = args.user.toLowerCase();
    await tx.order.upsert({
      where: {
        txHash_logIndex: {
          txHash: meta.transactionHash,
          logIndex: meta.logIndex,
        },
      },
      create: {
        type: "MINT",
        status: "COMPLETED",
        caliber: prismaCaliber,
        tokenAmount: args.tokenAmount.toString(),
        // usdcAmount unknown from finalization event alone -- leave null
        onChainOrderId: args.orderId.toString(),
        walletAddress: userAddress,
        txHash: meta.transactionHash,
        logIndex: meta.logIndex,
        chainId: CHAIN_ID,
        user: {
          connectOrCreate: {
            where: { walletAddress: userAddress },
            create: { walletAddress: userAddress },
          },
        },
      },
      update: {},
    });
  }

  // ActivityLog entry (keep existing logic but use the new fields)
  // ... (keep the existing ActivityLog block, but change order.amount to order.usdcAmount ?? order.tokenAmount ?? "0")
}
```

Note: The `MintFinalizedArgs` type does NOT include `usdcAmount` -- it only has `tokenAmount` and `priceUsed`. So on finalization we can only set `tokenAmount`. The `usdcAmount` was already set during `MintStarted`. If self-healing creates from finalization alone, `usdcAmount` will be null (acceptable -- partial data is better than no data).
